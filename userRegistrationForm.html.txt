function addUser()
{
	var listSize = document.getElementById("userList").children.length;
	listSize=listSize+3;

	var newUser = document.createElement("input");
	var br = document.createElement("br");
	newUser.type = "text";
	newUser.name = "username"+listSize;
	newUser.id = "username"+listSize;
	newUser.required = "true";
	var funVar = "validateUsername('username"+listSize+"')";
	newUser.setAttribute("onfocusout",funVar);
	document.getElementById("userList").appendChild(newUser); 
	
	var remove = document.createElement("input");
	remove.type = "button";
	remove.name="remove"+listSize;;
	remove.id="remove"+listSize;;
	remove.value="Remove";
	var removeUser = "removeUser("+listSize+")";
	remove.setAttribute("onClick",removeUser);
	document.getElementById("userList").appendChild(remove); 
	document.getElementById("userList").appendChild(br);
	document.getElementById("username"+listSize).focus();
	} 
	
	function removeUser(removeId)
	{
		document.getElementById("username"+removeId).remove();
		document.getElementById("remove"+removeId).remove();
		
	}
